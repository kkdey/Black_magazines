Introduction

We check on the word trends of the [black, negro] identity words.

Ebony words

dir <- list.files("../Ebony/")
library(wordVectors)
w2vec <- list()
words <- c()
for(l in 1:16){
  w2vec[[l]] <- read.vectors(paste0("../Ebony/", dir[l], "/pooled_word2vec.bin"))
  temp <- w2vec[[l]][which(nchar(rownames(w2vec[[l]])) > 2),]
  if(l==1){
    words <- rownames(temp)
  }else{
    words <- c(words, rownames(temp))
  }
}
